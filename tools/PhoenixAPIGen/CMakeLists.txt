# ── PhoenixAPIGen ──────────────────────────────────────────────────────────────
#
# Native executable that walks the live TypeRegistry and emits:
#   api.json — machine-readable description of all reflected types, properties,
#              methods, and constructors.
#
# Downstream consumers (PhoenixWasmGen, PhoenixLuaGen, tooling) use api.json to
# generate host_api.h, lua_bridge.c, IDE autocomplete stubs, etc.
#
# Linking model:
#   WHOLEARCHIVE forces every .obj from each static library to be included so
#   every PHX_DEFINE_TYPE static initializer runs and populates the TypeRegistry.

add_executable(PhoenixAPIGen
    "${CMAKE_CURRENT_LIST_DIR}/PhoenixAPIGen.cpp"
)

target_include_directories(PhoenixAPIGen PRIVATE
    "${CMAKE_SOURCE_DIR}/src"
)

target_link_libraries(PhoenixAPIGen PRIVATE
    PhoenixRTS
    PhoenixPhysics
    PhoenixSteering
    PhoenixSim
    nlohmann_json::nlohmann_json
)

if(MSVC)
    target_link_options(PhoenixAPIGen PRIVATE
        /WHOLEARCHIVE:$<TARGET_FILE:PhoenixRTS>
        /WHOLEARCHIVE:$<TARGET_FILE:PhoenixPhysics>
        /WHOLEARCHIVE:$<TARGET_FILE:PhoenixSteering>
        /WHOLEARCHIVE:$<TARGET_FILE:PhoenixSim>
    )
elseif(APPLE)
    target_link_options(PhoenixAPIGen PRIVATE
        -Wl,-force_load,$<TARGET_FILE:PhoenixRTS>
        -Wl,-force_load,$<TARGET_FILE:PhoenixPhysics>
        -Wl,-force_load,$<TARGET_FILE:PhoenixSteering>
        -Wl,-force_load,$<TARGET_FILE:PhoenixSim>
    )
else()
    target_link_options(PhoenixAPIGen PRIVATE
        -Wl,--whole-archive
        $<TARGET_FILE:PhoenixRTS>
        $<TARGET_FILE:PhoenixPhysics>
        $<TARGET_FILE:PhoenixSteering>
        $<TARGET_FILE:PhoenixSim>
        -Wl,--no-whole-archive
    )
endif()

set_property(TARGET PhoenixAPIGen PROPERTY FOLDER "Tools")

# ── api.json generation ────────────────────────────────────────────────────────
#
# Run PhoenixAPIGen after it is built to dump api.json into the build root.
# Downstream steps (PhoenixWasmGen, PhoenixLuaGen, etc.) depend on this target
# so that api.json is always up-to-date before code generation begins.
#
# PHOENIX_API_JSON is cached as an INTERNAL variable so that other
# CMakeLists.txt files can reference the path without hard-coding it.

set(PHOENIX_API_JSON "${CMAKE_BINARY_DIR}/api.json" CACHE INTERNAL
    "Path to the api.json file generated by PhoenixAPIGen")

add_custom_command(
    OUTPUT  "${PHOENIX_API_JSON}"
    COMMAND $<TARGET_FILE:PhoenixAPIGen> --output "${PHOENIX_API_JSON}"
    DEPENDS PhoenixAPIGen
    COMMENT "PhoenixAPIGen: writing ${PHOENIX_API_JSON}"
)

add_custom_target(GenerateAPI ALL
    DEPENDS "${PHOENIX_API_JSON}"
)

set_property(TARGET GenerateAPI PROPERTY FOLDER "Tools")
